## JS Architecture Overview

```
DATA (inline JSON const)
  ↓
computePositions()   → positions: { [id]: {x, y} }  — computed once at load
  ↓
render()             → clears SVG, draws visible edges then visible nodes
  ↑
state: hiddenNodes (Set), hiddenDomains (Set), highlightedNode (string|null)

Event sources:
  domain checkboxes → toggleDomain(domain, bool)
  node click        → hideNode(id)       — cascade hides isolated nodes
  sidebar click     → highlightNode(id)  — scrolls SVG, accents border
  reset button      → resetState()
```

- [ ] **Step 2: Add the layout algorithm**

```js
function computePositions() {
  // Group tables by domain, preserving DOMAIN_ORDER
  const byDomain = {};
  for (const d of DOMAIN_ORDER) byDomain[d] = [];
  for (const t of DATA.tables) {
    if (!byDomain[t.domain]) byDomain[t.domain] = [];
    byDomain[t.domain].push(t);
  }

  let x = PAD;
  for (const domain of DOMAIN_ORDER) {
    const tables = byDomain[domain] || [];
    if (!tables.length) continue;
    let y = PAD;
    for (const t of tables) {
      positions[t.id] = { x, y };
      y += NODE_H + ROW_GAP;
    }
    x += NODE_W + COL_GAP;
  }
}
```

- [ ] **Step 3: Add SVG size helper**

```js
function setSVGSize() {
  const svg = document.getElementById('diagram');
  let maxX = 0, maxY = 0;
  for (const { x, y } of Object.values(positions)) {
    maxX = Math.max(maxX, x + NODE_W + PAD);
    maxY = Math.max(maxY, y + NODE_H + PAD);
  }
  svg.setAttribute('width', maxX);
  svg.setAttribute('height', maxY);
}
```

```js
function collectEdges() {
  const seen = new Set();
  const edges = [];
  for (const table of DATA.tables) {
    for (const rel of (table.relations || [])) {
      const targets = rel.targets
        ? rel.targets.filter(t => t !== '*')
        : rel.target ? [rel.target] : [];
      for (const target of targets) {
        // Deduplicate: use sorted pair as key
        const key = [table.id, target].sort().join('|') + '|' + rel.type;
        if (seen.has(key)) continue;
        seen.add(key);
        edges.push({
          from: table.id,
          to: target,
          type: rel.type,
          label: rel.label || rel.type,
        });
      }
    }
  }
  return edges;
}
```

```js
function renderEdges(visibleIds) {
  const svg = document.getElementById('diagram');
  svg.querySelectorAll('.edge').forEach(el => el.remove());

  const visible = new Set(visibleIds);
  const edges = collectEdges();

  for (const edge of edges) {
    if (!positions[edge.from] || !positions[edge.to]) continue;
    const bothVisible = visible.has(edge.from) && visible.has(edge.to);

    const fromPos = positions[edge.from];
    const toPos = positions[edge.to];

    // Center of right edge of source, center of left edge of target
    const x1 = fromPos.x + NODE_W;
    const y1 = fromPos.y + NODE_H / 2;
    const x2 = toPos.x;
    const y2 = toPos.y + NODE_H / 2;

    // Bezier control points
    const cx = (x1 + x2) / 2;

    const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');
    g.classList.add('edge');
    if (!bothVisible) g.classList.add('hidden');

    const path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
    path.setAttribute('d', `M ${x1} ${y1} C ${cx} ${y1}, ${cx} ${y2}, ${x2} ${y2}`);
    path.setAttribute('fill', 'none');
    path.setAttribute('stroke', '#bbb');
    path.setAttribute('stroke-width', '1.5');

    // Style by relation type
    if (edge.type === 'extends') {
      path.setAttribute('stroke-dasharray', '6,3');
      path.setAttribute('marker-end', 'url(#arrow)');
    } else if (edge.type === 'many_to_many') {
      path.setAttribute('marker-end', 'url(#arrow)');
      path.setAttribute('marker-start', 'url(#arrow-start)');
    } else if (edge.type === 'polymorphic') {
      path.setAttribute('stroke-dasharray', '2,3');
      path.setAttribute('marker-end', 'url(#arrow)');
    } else {
      path.setAttribute('marker-end', 'url(#arrow)');
    }

    g.appendChild(path);

    // Edge label (mid-point)
    const label = document.createElementNS('http://www.w3.org/2000/svg', 'text');
    label.setAttribute('x', cx);
    label.setAttribute('y', (y1 + y2) / 2 - 4);
    label.setAttribute('font-size', '9');
    label.setAttribute('text-anchor', 'middle');
    label.setAttribute('fill', '#999');
    label.textContent = edge.label;
    g.appendChild(label);

    // Insert edges BEFORE nodes so nodes render on top
    const firstNode = svg.querySelector('.node');
    svg.insertBefore(g, firstNode);
  }
}
```

## Task 7: Node Click Hiding with Cascade

**Files:**
- Modify: `20_projects/system/docs/architecture/db-viewer.html`

- [ ] **Step 1: Add cascade hide algorithm**

```js
function cascadeHide(clickedId) {
  hiddenNodes.add(clickedId);

  // Build adjacency list for currently visible nodes
  const visible = new Set(
    DATA.tables
      .filter(t => !hiddenDomains.has(t.domain) && !hiddenNodes.has(t.id))
      .map(t => t.id)
  );

  let changed = true;
  while (changed) {
    changed = false;
    for (const t of DATA.tables) {
      if (!visible.has(t.id)) continue;
      // Collect all neighbors of t in the full graph
      const neighbors = getNeighbors(t.id);
      // If ALL neighbors are hidden, this node is isolated → hide it
      if (neighbors.length > 0 && neighbors.every(n => !visible.has(n))) {
        visible.delete(t.id);
        hiddenNodes.add(t.id);
        changed = true;
      }
    }
  }
}

function getNeighbors(id) {
  const neighbors = new Set();
  for (const t of DATA.tables) {
    for (const rel of (t.relations || [])) {
      const targets = rel.targets
        ? rel.targets.filter(x => x !== '*')
        : rel.target ? [rel.target] : [];
      if (t.id === id) {
        targets.forEach(tgt => neighbors.add(tgt));
      }
      if (targets.includes(id)) {
        neighbors.add(t.id);
      }
    }
  }
```
